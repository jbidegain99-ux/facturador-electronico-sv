  /**
   * Check if a token is still valid
   */
  private isTokenValid(tokenInfo: TokenInfo): boolean {
    // Give 5 minute buffer before expiry
    const bufferMs = 5 * 60 * 1000;
    const now = Date.now();
    return tokenInfo.expiresAt.getTime() - bufferMs > now;
  }
